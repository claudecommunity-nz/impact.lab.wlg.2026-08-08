# Walking Skeleton Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Prove the entire golden path of PLAN.md v4.1 end-to-end with minimal features: `pnpm new-module` scaffold → Python loader publishes a signal → module tile + signal appear live in a bare Next.js dashboard, against token-gated Supabase (local first, then the cloud project).

**Architecture:** pnpm monorepo (apps/dashboard, packages/plugin-sdk, packages/shared) + uv workspace (packages/wcc-impact-platform-py, modules/*/loader). Modules are folders with a `module.config.ts` manifest discovered by codegen into a gitignored registry. All writes to Supabase require an `x-event-token` header checked by RLS; the `enabled` column is protected by column-level grants so only the service role can flip it.

**Tech Stack:** Next.js 15 (App Router), React 19, @supabase/supabase-js v2, zod, vitest, tsx, Python 3.12, httpx, pytest, Supabase CLI (local stack via Docker), uv 0.5.x, pnpm 10.

## Global Constraints

- Next.js pinned to major 15 (PLAN.md §3.2); lockfile is the pin.
- pnpm workspace globs: `apps/*`, `packages/*`, `modules/*` (PLAN.md §6).
- uv workspace members: `packages/wcc-impact-platform-py`, `modules/*/loader` (PLAN.md §6).
- `apps/dashboard/registry.gen.ts` is gitignored and regenerated on every dev/build/CI run — never committed, never hand-edited (PLAN.md §3.2). Already in `.gitignore`.
- Module UIs import only from `@wcc-impact/plugin-sdk` and React (PLAN.md §5).
- Every write policy requires the event token header `x-event-token` (PLAN.md §7). Signals INSERT additionally requires the target module to be registered **and `enabled`**.
- `enabled` column on `modules`: changeable only by service role (PLAN.md §7.1).
- No secrets in any committed file. Local-stack demo credentials (fixed, publicly documented defaults shipped by the Supabase CLI) and the local-only dev token `dev-event-token` in `supabase/seed.sql` are not secrets. The cloud event token and DB password live only in gitignored `.env` (PLAN.md §7.5).
- Loaders are Python; UI is TypeScript (PLAN.md §8).
- Module folder `modules/_template` is excluded from codegen and scaffolded copies replace the literal string `template` (it is valid, compiling code — no `{{PLACEHOLDER}}` tokens).
- Working directory for all commands: repo root `/Users/adamholt/ClaudeCommunityNZ/impact.lab.wlg.2026-08-08` unless stated.

- [ ] **Step 1: Write `supabase/migrations/20260722000001_core.sql`**

```sql
-- Core tables + event-token RLS (PLAN.md sections 7.1-7.3)

create schema if not exists private;

create table private.event_config (
  id int primary key default 1 check (id = 1),
  token text not null
);
-- no grants: anon/authenticated cannot read this table

create or replace function private.has_event_token()
returns boolean
language sql stable security definer
set search_path = ''
as $$
  select coalesce(
    current_setting('request.headers', true)::json ->> 'x-event-token', ''
  ) = coalesce((select token from private.event_config where id = 1), '<unset>')
$$;

grant usage on schema private to anon, authenticated;
grant execute on function private.has_event_token() to anon, authenticated;

create table public.modules (
  id           text primary key check (id ~ '^[a-z0-9][a-z0-9-]{2,49}$'),
  name         text not null check (char_length(name) between 1 and 100),
  icon         text check (char_length(icon) <= 16),
  description  text check (char_length(description) <= 500),
  problem      int check (problem between 1 and 5),
  enabled      boolean not null default true,
  last_seen    timestamptz,
  updated_at   timestamptz not null default now()
);

create table public.signals (
  id           uuid primary key default gen_random_uuid(),
  created_at   timestamptz not null default now(),
  observed_at  timestamptz,
  source       text check (char_length(source) <= 200),
  source_type  text check (source_type in ('official','community','media','sensor')),
  signal_type  text not null check (char_length(signal_type) between 1 and 60),
  title        text not null check (char_length(title) between 1 and 200),
  description  text check (char_length(description) <= 4000),
  lat          double precision check (lat between -90 and 90),
  lng          double precision check (lng between -180 and 180),
  place_name   text check (char_length(place_name) <= 200),
  severity     text check (severity in ('minor','moderate','severe','extreme')),
  verification text not null default 'unverified'
               check (verification in ('unverified','probable','confirmed','false')),
  confidence   real check (confidence between 0 and 1),
  link         text check (char_length(link) <= 500),
  media_urls   text[] check (coalesce(array_length(media_urls, 1), 0) <= 10),
  module_id    text not null references public.modules(id),
  raw          jsonb
);

create index signals_created_at_idx on public.signals (created_at desc);
create index signals_module_id_idx on public.signals (module_id);

alter table public.modules enable row level security;
alter table public.signals enable row level security;

-- modules: everyone reads; room (token) registers/updates; `enabled` is
-- protected by COLUMN grants — anon simply has no privilege on it.
revoke all on public.modules from anon, authenticated;
grant select on public.modules to anon, authenticated;
grant insert (id, name, icon, description, problem) on public.modules to anon, authenticated;
grant update (name, icon, description, problem, last_seen) on public.modules to anon, authenticated;

create policy modules_select on public.modules
  for select to anon, authenticated using (true);
create policy modules_insert on public.modules
  for insert to anon, authenticated with check (private.has_event_token());
create policy modules_update on public.modules
  for update to anon, authenticated
  using (private.has_event_token()) with check (private.has_event_token());

-- signals: everyone reads; room inserts, only for registered+enabled modules
revoke all on public.signals from anon, authenticated;
grant select on public.signals to anon, authenticated;
grant insert on public.signals to anon, authenticated;
grant update (verification) on public.signals to authenticated;

create policy signals_select on public.signals
  for select to anon, authenticated using (true);
create policy signals_insert on public.signals
  for insert to anon, authenticated
  with check (
    private.has_event_token()
    and exists (
      select 1 from public.modules m
      where m.id = module_id and m.enabled
    )
  );
create policy signals_verify on public.signals
  for update to authenticated
  using (private.has_event_token()) with check (private.has_event_token());

alter publication supabase_realtime add table public.signals;
alter publication supabase_realtime add table public.modules;
```

from datetime import datetime, timezone
from typing import Any, Callable

from . import client

__all__ = ["register_module", "publish_signal", "heartbeat", "run_every", "client"]

MIN_INTERVAL_SECONDS = 5  # PLAN.md section 8: one hot loop must not flood the room


def register_module(
    module_id: str,
    name: str,
    icon: str | None = None,
    description: str | None = None,
    problem: int | None = None,
) -> None:
    """Upsert this module's registry row. Tile appears on the dashboard from here."""
    row = {"id": module_id, "name": name, "icon": icon,
           "description": description, "problem": problem}
    client._post(
        "/modules?on_conflict=id",
        {k: v for k, v in row.items() if v is not None},
        extra_prefer="resolution=merge-duplicates",
    )


def publish_signal(module_id: str, title: str, signal_type: str, **fields: Any) -> dict:
    """Insert one signal. Requires the module to be registered and enabled."""
    body = {"module_id": module_id, "title": title, "signal_type": signal_type, **fields}
    return client._post("/signals", body)[0]


def heartbeat(module_id: str) -> None:
    """Stamp modules.last_seen so the health strip shows the loader alive."""
    now = datetime.now(timezone.utc).isoformat()
    client._patch(f"/modules?id=eq.{module_id}", {"last_seen": now})


def run_every(seconds: float, fn: Callable[[], None]) -> None:
    """Run fn forever at a fixed cadence. Floor of 5 s protects the shared feed."""
    if seconds < MIN_INTERVAL_SECONDS:
        raise ValueError(
            f"run_every interval must be >= {MIN_INTERVAL_SECONDS}s "
            "(shared-platform politeness, PLAN.md section 8)"
        )
    while True:
        started = time.monotonic()
        fn()
        elapsed = time.monotonic() - started
        time.sleep(max(0.0, seconds - elapsed))

import { existsSync, readdirSync } from 'node:fs';
import { join } from 'node:path';

export function discoverModules(repoRoot: string): string[] {
  const dir = join(repoRoot, 'modules');
  if (!existsSync(dir)) return [];
  return readdirSync(dir, { withFileTypes: true })
    .filter((d) => d.isDirectory() && !d.name.startsWith('_'))
    .filter((d) => existsSync(join(dir, d.name, 'module.config.ts')))
    .map((d) => d.name)
    .sort();
}

import type { Signal } from '@wcc-impact/shared';

export type SignalFilter = {
  moduleId?: string;
  signalType?: string;
  since?: string; // ISO timestamp, inclusive
};

export function filterSignals(signals: Signal[], filter?: SignalFilter): Signal[] {
  if (!filter) return signals;
  return signals.filter(
    (s) =>
      (!filter.moduleId || s.module_id === filter.moduleId) &&
      (!filter.signalType || s.signal_type === filter.signalType) &&
      (!filter.since || (s.created_at ?? '') >= filter.since),
  );
}

'use client';
import { createContext, useContext, useEffect, useState, type ReactNode } from 'react';
import type { ModuleRow, Signal } from '@wcc-impact/shared';
import { filterSignals, type SignalFilter } from './filter';
import { supabase } from './supabase';

type PlatformState = { signals: Signal[]; modules: ModuleRow[] };
const PlatformContext = createContext<PlatformState>({ signals: [], modules: [] });

export function PlatformProvider({ children }: { children: ReactNode }) {
  const [signals, setSignals] = useState<Signal[]>([]);
  const [modules, setModules] = useState<ModuleRow[]>([]);

  useEffect(() => {
    supabase
      .from('signals')
      .select('*')
      .order('created_at', { ascending: false })
      .limit(200)
      .then(({ data }) => setSignals((data as Signal[]) ?? []));
    supabase
      .from('modules')
      .select('*')
      .then(({ data }) => setModules((data as ModuleRow[]) ?? []));

    const channel = supabase
      .channel('platform')
      .on(
        'postgres_changes',
        { event: 'INSERT', schema: 'public', table: 'signals' },
        (payload) => setSignals((prev) => [payload.new as Signal, ...prev].slice(0, 500)),
      )
      .on(
        'postgres_changes',
        { event: '*', schema: 'public', table: 'modules' },
        (payload) =>
          setModules((prev) => {
            const row = payload.new as ModuleRow;
            const rest = prev.filter((m) => m.id !== row.id);
            return [...rest, row].sort((a, b) => a.id.localeCompare(b.id));
          }),
      )
      .subscribe();
    return () => {
      void supabase.removeChannel(channel);
    };
  }, []);

  return <PlatformContext.Provider value={{ signals, modules }}>{children}</PlatformContext.Provider>;
}

export function useSignals(filter?: SignalFilter): Signal[] {
  return filterSignals(useContext(PlatformContext).signals, filter);
}

export function useModules(): ModuleRow[] {
  return useContext(PlatformContext).modules.filter((m) => m.enabled);
}

- [ ] **Step 4: Write `scripts/lib/scaffold.ts`**

```ts
import { cpSync, existsSync, readFileSync, readdirSync, statSync, writeFileSync } from 'node:fs';
import { join } from 'node:path';

const ID_RE = /^[a-z0-9][a-z0-9-]{2,49}$/;

function titleCase(id: string): string {
  return id.split('-').map((w) => w[0].toUpperCase() + w.slice(1)).join(' ');
}

function* walk(dir: string): Generator<string> {
  for (const entry of readdirSync(dir)) {
    const p = join(dir, entry);
    if (statSync(p).isDirectory()) yield* walk(p);
    else yield p;
  }
}

export function scaffoldModule(repoRoot: string, id: string): string {
  if (!ID_RE.test(id)) {
    throw new Error(`module id must match ${ID_RE} (lowercase, digits, hyphens), got "${id}"`);
  }
  const target = join(repoRoot, 'modules', id);
  if (existsSync(target)) throw new Error(`modules/${id} already exists`);

  cpSync(join(repoRoot, 'modules', '_template'), target, { recursive: true });

  const name = titleCase(id);
  const pyName = titleCase(id).replace(/ /g, '');
  for (const file of walk(target)) {
    const before = readFileSync(file, 'utf8');
    const after = before
      .replaceAll('hack-template', id)
      .replaceAll('Hack Template', name)
      .replaceAll('HackTemplateUi', `${pyName}Ui`);
    if (after !== before) writeFileSync(file, after);
  }
  return target;
}
```
